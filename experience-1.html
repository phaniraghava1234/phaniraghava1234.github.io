// Overview

Developed MAGNUS Orbito, a Python and Orekit port and extension of a legacy Excel workbook for early-phase space mission analysis. This tool supports single-satellite workflows, parametric Walker constellations, and TLE-based workflows via interactive Jupyter notebooks. By integrating the Orekit space-dynamics library using JPype1, I enabled high-fidelity orbit propagation, global coverage mapping, and sensor footprint geometry, replacing static spreadsheets with robust Python analytics and interactive Plotly maps.

"Built a unified Python pipeline for high-fidelity Eckstein-Hechler propagation, complex visibility calculations, and global grid coverage using Orekit — turning a manual spreadsheet workflow into a validated, interactive software toolkit."

def propagate_orbit(
    orbit, start, end, step_seconds: float,
    propagator_type: Literal['keplerian', 'eckstein-hechler'] = 'eckstein-hechler',
) -> pd.DataFrame:
    
    eme2000 = FramesFactory.getEME2000()
    itrf    = FramesFactory.getITRF(IERSConventions.IERS_2010, True)
    earth   = OneAxisEllipsoid(
                  Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
                  Constants.WGS84_EARTH_FLATTENING, itrf)

    propagator = _make_propagator(orbit, propagator_type)

    rows = []
    t = start_abs
    while t.compareTo(end_abs) <= 0:
        state = propagator.propagate(t)
        pv  = state.getPVCoordinates(eme2000)
        pos = pv.getPosition()
        vel = pv.getVelocity()
        gp  = earth.transform(pos, eme2000, t)

        rows.append({
            'epoch'  : to_datetime(t),
            'x'      : pos.getX(),  'y'  : pos.getY(),  'z'  : pos.getZ(),
            'lat_deg': math.degrees(gp.getLatitude()),
            'lon_deg': math.degrees(gp.getLongitude()),
            'alt_km' : gp.getAltitude() / 1e3,
        })
        t = t.shiftedBy(float(step_seconds))

    return pd.DataFrame(rows).set_index('epoch')

Sample of the global grid coverage metric calculation algorithm for constellation analysis: 

def compute_grid_metrics(prop_dict, grid_points, start, end, min_el_deg=5.0):
    span_s = (end - start).total_seconds()
    rows   = []

    for pt in grid_points:
        pt.min_elevation_deg = min_el_deg
        frames = []
        for df in prop_dict.values():
            w = compute_access_windows(df, [pt])
            if not w.empty: frames.append(w)

        if not frames:
            rows.append(_empty_row(pt.lat_deg, pt.lon_deg))
            continue

        combined = (pd.concat(frames, ignore_index=True)
                      .sort_values('start').reset_index(drop=True))
        n = len(combined)

        merged_s = _merge_duration(combined['start'].tolist(), combined['end'].tolist())
        cov_pct  = 100.0 * merged_s / span_s if span_s > 0 else 0.0

        gaps = [(combined.loc[i, 'start'] - combined.loc[i-1, 'end']).total_seconds()
                for i in range(1, n)
                if (combined.loc[i, 'start'] - combined.loc[i-1, 'end']).total_seconds() > 0]

        rows.append({
            'lat_deg':       pt.lat_deg,
            'lon_deg':       pt.lon_deg,
            'n_contacts':    n,
            'coverage_pct':  cov_pct,
            'mean_gap_s':    float(np.mean(gaps))           if gaps else float('nan'),
            'P95_gap_s':     float(np.percentile(gaps, 95)) if gaps else float('nan'),
            'max_gap_s':     float(np.max(gaps))            if gaps else float('nan'),
        })

    return pd.DataFrame(rows)

Sample of the mission analysis logic calculating data downlink latencies and revisit gap CDFs: 

def compute_data_latency(obs_windows, gsn_windows):
    out_cols = ['satellite', 'obs_target', 'obs_start', 'obs_end',
                'gsn_station', 'gsn_contact_start', 'latency_s']
    if obs_windows.empty or gsn_windows.empty: return pd.DataFrame(columns=out_cols)

    rows = []
    for _, obs in obs_windows.iterrows():
        sat     = obs['satellite']
        obs_end = obs['end']
        cands   = gsn_windows[
            (gsn_windows['satellite'] == sat) &
            (gsn_windows['start']    >= obs_end)
        ]
        if cands.empty:
            rows.append({'satellite': sat, 'obs_target': obs['target'],
                         'obs_start': obs['start'], 'obs_end': obs_end,
                         'gsn_station': None, 'gsn_contact_start': pd.NaT,
                         'latency_s': float('nan')})
        else:
            nxt = cands.loc[cands['start'].idxmin()]
            rows.append({'satellite': sat, 'obs_target': obs['target'],
                         'obs_start': obs['start'], 'obs_end': obs_end,
                         'gsn_station': nxt['target'],
                         'gsn_contact_start': nxt['start'],
                         'latency_s': (nxt['start'] - obs_end).total_seconds()})
    return pd.DataFrame(rows, columns=out_cols)

def compute_revisit_cdf(windows, target=None):
    df = windows.copy()
    if target is not None: df = df[df['target'] == target]
    df = df.sort_values('start').reset_index(drop=True)
    if len(df) < 2: return np.array([]), np.array([])

    gaps = [(df.loc[i, 'start'] - df.loc[i-1, 'end']).total_seconds()
            for i in range(1, len(df))
            if (df.loc[i, 'start'] - df.loc[i-1, 'end']).total_seconds() > 0]
    if not gaps: return np.array([]), np.array([])
    gaps_s = np.sort(gaps)
    return gaps_s, np.arange(1, len(gaps_s)+1) / len(gaps_s)
